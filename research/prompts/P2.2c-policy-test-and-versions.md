### API Client Methods

Add:

```typescript
async testPolicy(data: { agent_id: string; operation: string; target_integration: string; resource_scope: string; data_classification: string }) {
  return this.post('/api/v1/policies/test', data);
}

async getPolicyVersions(policyId: string, params?: { limit?: number; offset?: number }) {
  const query = new URLSearchParams();
  if (params?.limit) query.set('limit', String(params.limit));
  if (params?.offset) query.set('offset', String(params.offset));
  return this.get(`/api/v1/policies/${policyId}/versions?${query}`);
}
```
