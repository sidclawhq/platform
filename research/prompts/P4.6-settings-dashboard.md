### 8. API Client Methods

Add:

```typescript
async getTenantSettings() {
  return this.get('/api/v1/tenant/settings');
}

async updateTenantSettings(data: { name?: string; settings?: Record<string, unknown> }) {
  return this.patch('/api/v1/tenant/settings', data);
}

async listWebhooks() {
  return this.get('/api/v1/webhooks');
}

async createWebhook(data: { url: string; events: string[]; description?: string }) {
  return this.post('/api/v1/webhooks', data);
}

async deleteWebhook(id: string) {
  return this.delete(`/api/v1/webhooks/${id}`);
}

async testWebhook(id: string) {
  return this.post(`/api/v1/webhooks/${id}/test`, {});
}

async getWebhookDeliveries(id: string, params?: { status?: string; limit?: number }) {
  const query = new URLSearchParams();
  if (params?.status) query.set('status', params.status);
  if (params?.limit) query.set('limit', String(params.limit));
  return this.get(`/api/v1/webhooks/${id}/deliveries?${query}`);
}
```
