### 6. Relative Time Utility

Create `apps/dashboard/src/lib/format.ts`:

```typescript
export function relativeTime(date: string | Date): string {
  const now = Date.now();
  const then = new Date(date).getTime();
  const seconds = Math.floor((now - then) / 1000);

  if (seconds < 60) return 'just now';
  if (seconds < 3600) return `${Math.floor(seconds / 60)} min ago`;
  if (seconds < 86400) return `${Math.floor(seconds / 3600)}h ago`;
  return `${Math.floor(seconds / 86400)}d ago`;
}
```

### 7. API Client Methods

```typescript
// In api-client.ts or a new file lib/api/approvals.ts

async listApprovals(params?: { status?: string; limit?: number; offset?: number }) {
  const query = new URLSearchParams();
  if (params?.status) query.set('status', params.status);
  if (params?.limit) query.set('limit', String(params.limit));
  if (params?.offset) query.set('offset', String(params.offset));
  return this.get<{ data: any[]; pagination: { total: number; limit: number; offset: number } }>(
    `/api/v1/approvals?${query}`
  );
}

async getApproval(id: string) {
  return this.get<ApprovalWithContext>(`/api/v1/approvals/${id}`);
}

async approveRequest(id: string, body: { approver_name: string; decision_note?: string }) {
  return this.post<ApprovalWithContext>(`/api/v1/approvals/${id}/approve`, body);
}

async denyRequest(id: string, body: { approver_name: string; decision_note?: string }) {
  return this.post<ApprovalWithContext>(`/api/v1/approvals/${id}/deny`, body);
}
```
