#### Email/Password (`apps/api/src/auth/providers/email-password.ts`)

```typescript
import { hash, compare } from 'bcrypt';

const BCRYPT_ROUNDS = 12;

export class EmailPasswordProvider {
  async hashPassword(password: string): Promise<string> {
    return hash(password, BCRYPT_ROUNDS);
  }

  async verifyPassword(password: string, hash: string): Promise<boolean> {
    return compare(password, hash);
  }

  validatePassword(password: string): { valid: boolean; message?: string } {
    if (password.length < 8) return { valid: false, message: 'Password must be at least 8 characters' };
    if (password.length > 128) return { valid: false, message: 'Password must be at most 128 characters' };
    return { valid: true };
  }

  validateEmail(email: string): boolean {
    // Basic format check — not full RFC 5322
    return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
  }
}
```

**Signup/first-login flow (same for all providers):**

```typescript
async function provisionNewUser(email: string, name: string, authProvider: string, authProviderId: string | null, passwordHash: string | null) {
  return prisma.$transaction(async (tx) => {
    // 1. Check if user already exists
    const existing = await tx.user.findUnique({ where: { email } });
    if (existing) {
      throw new ConflictError('An account with this email already exists');
    }

    // 2. Create tenant
    const tenantName = `${name}'s workspace`;
    const tenantSlug = email.split('@')[0].toLowerCase().replace(/[^a-z0-9]/g, '-') + '-' + Date.now().toString(36);
    const tenant = await tx.tenant.create({
      data: {
        name: tenantName,
        slug: tenantSlug,
        plan: 'free',
        settings: {
          default_approval_ttl_seconds: 86400,
          default_data_classification: 'internal',
          notification_email: email,
          notifications_enabled: true,
        },
        onboarding_state: {
          copy_api_key: false,
          register_agent: false,
          create_policy: false,
          run_evaluation: false,
          see_trace: false,
        },
      },
    });

    // 3. Create user (admin role — first user in tenant)
    const user = await tx.user.create({
      data: {
        tenant_id: tenant.id,
        email,
        name,
        role: 'admin',
        auth_provider: authProvider,
        auth_provider_id: authProviderId,
        password_hash: passwordHash,
      },
    });

    // 4. Create default API key
    const { createHash, randomBytes } = await import('crypto');
    const rawKey = 'ai_' + randomBytes(32).toString('hex');
    const keyHash = createHash('sha256').update(rawKey).digest('hex');

    await tx.apiKey.create({
      data: {
        tenant_id: tenant.id,
        name: 'Default Key',
        key_prefix: rawKey.substring(0, 12),
        key_hash: keyHash,
        scopes: ['evaluate', 'traces:read', 'traces:write', 'approvals:read'],
      },
    });

    return { user, tenant, apiKey: rawKey };
  });
}
```

**For returning users (OAuth re-login):**
- If user exists with matching email → create session (don't create new tenant)
- If user exists with different auth_provider → return error "Account exists with different provider"

### 4. Free Plan Limits Enforcement

Create `apps/api/src/middleware/plan-limits.ts`:

```typescript
const FREE_PLAN_LIMITS = {
  max_agents: 5,
  max_policies_per_agent: 10,
  max_api_keys: 2,
  max_webhook_endpoints: 1,
  max_users: 3,
  trace_retention_days: 7,
  rate_limit_evaluate_per_min: 100,
};

const TEAM_PLAN_LIMITS = { /* higher limits */ };
const ENTERPRISE_PLAN_LIMITS = { /* unlimited for most */ };

export async function checkPlanLimit(
  prisma: PrismaClient,
  tenantId: string,
  limitName: keyof typeof FREE_PLAN_LIMITS,
  currentCount: number
): Promise<void> {
  const tenant = await prisma.tenant.findUnique({ where: { id: tenantId }, select: { plan: true } });
  const limits = tenant?.plan === 'enterprise' ? ENTERPRISE_PLAN_LIMITS
    : tenant?.plan === 'team' ? TEAM_PLAN_LIMITS
    : FREE_PLAN_LIMITS;

  const max = limits[limitName];
  if (max !== undefined && currentCount >= max) {
    throw new PlanLimitError(limitName, currentCount, max);
  }
}
```

**Add `PlanLimitError` to API errors:**

```typescript
export class PlanLimitError extends AppError {
  constructor(limitName: string, current: number, max: number) {
    super('plan_limit_reached', 402,
      `Free plan allows up to ${max} ${limitName.replace(/_/g, ' ')}. Upgrade to Team for more.`,
      { limit: limitName, current, max }
    );
  }
}
```

**Enforce in routes:**
- `POST /api/v1/agents` → check `max_agents`
- `POST /api/v1/policies` → check `max_policies_per_agent`
- `POST /api/v1/api-keys` → check `max_api_keys`
- `POST /api/v1/webhooks` → check `max_webhook_endpoints`
- User creation (via OIDC first login) → check `max_users`
