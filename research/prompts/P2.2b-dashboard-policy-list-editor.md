### API Client Methods

Add to the API client:

```typescript
async listPolicies(params?: { agent_id?: string; effect?: string; data_classification?: string; search?: string; limit?: number; offset?: number }) {
  const query = new URLSearchParams();
  for (const [key, value] of Object.entries(params ?? {})) {
    if (value !== undefined && value !== '') query.set(key, String(value));
  }
  return this.get(`/api/v1/policies?${query}`);
}

async getPolicy(id: string) {
  return this.get(`/api/v1/policies/${id}`);
}

async createPolicy(data: PolicyRuleCreateInput) {
  return this.post('/api/v1/policies', data);
}

async updatePolicy(id: string, data: Partial<PolicyRuleUpdateInput>) {
  return this.patch(`/api/v1/policies/${id}`, data);
}

async deletePolicy(id: string) {
  return this.delete(`/api/v1/policies/${id}`);
}
```
